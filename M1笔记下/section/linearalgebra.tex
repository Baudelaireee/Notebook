\subsection{Semi-direct product}

\begin{definition}
    Let \(G\) and \(H\) be groups, and let \(\varphi: H \to \Aut(G)\) be a group homomorphism. The \textbf{semi-direct product} \(G \rtimes_\varphi H\) is the set \(G \times H\) with the multiplication defined by
    \[(g_1, h_1)(g_2, h_2) = (g_1 \varphi(h_1)(g_2), h_1 h_2)\]
    for all \(g_1, g_2 \in G\) and \(h_1, h_2 \in H\).

\end{definition}

\begin{remark}
    If \(\varphi\) is trival, i.e. \(\varphi(h) = \mathrm{id}_G\) for all \(h \in H\), then the semi-direct product \(G \rtimes_\varphi H\) is just the \textbf{direct product} \(G \times H\).

    Conversely, if \(G \rtimes H\) is abelian, then \(\varphi\) must be trivial by caculating \((1,h)(g,1) = (g,1)(1,h)\), since it will be \(\varphi(h)(g) = g\) for any \(h\) and \(g\).

\end{remark}

\end{lemma}

\begin{definition}
    A group \(G\) is an inner semi-direct product of \(N\) and \(H\) if \(N \vartriangleleft G\), \(H < G\) and \(G = NH\) with \(N \cap H = \{e\}\).
\end{definition}

\begin{remark}
    The definition implies that the group has semi-direct product structure, and the automorphism comes from the inner automorphism:
    \[G \cong N \rtimes_{\varphi} H, \quad\]
    where \(\varphi: H \to \Aut(N)\) is defined by \(\varphi(h)(n) = hnh^{-1}\), it is well-defined since \(N\) is normal. To prove the isomorphism, we set \(f:N \rtimes H \to G\) by \(f(n,h) = nh\), then it is surjective since \(NH = G\), and the it is indeed a morphism
    \begin{align*}
        f((n_1,h_1)(n_2,h_2)) &= f(n_1 \varphi(h_1)(n_2), h_1 h_2)\\
        &= n_1 h_1 n_2 h_1^{-1} h_1 h_2\\
        &= n_1 h_1 n_2 h_2\\
        &= f(n_1,h_1)f(n_2,h_2)
    \end{align*}
    So it is enough to check the injectivity: if \(f(n,h) = e\), then \(n = h^{-1} \in N \cap H\), thus \(n = e\) and \(h = e\).

\end{remark}


\begin{proposition}
    Let \(G \rtimes H\) be a semi-direct product with respect to \(\varphi: H \to \Aut(G)\), then \(G\) can be idetified as  a normal subgroup of \(G \rtimes H\) while \(H\) can be identified as a subgroup of \(G \rtimes H\), and we have a splitting exact sequence
\[0 \to G \to G \rtimes H \to H \to 0\]
\end{proposition}

\begin{proof}
    We consider the inclusion \(i:G \to G \rtimes H\) defined by \(i(g) = (g,1)\) and the projection \(\pi:G \rtimes H \to H\) defined by \(\pi(g,h) = h\), then we can verify  that \(i\) and \(\pi\) are homomorphisms, and \(\ker \pi = i(G) \cong G\), so it shows that \(G\) can be identified as a normal subgroup of \(G \rtimes H\) and the sequence is exact. Finally, to prove that the sequence splits, we define a section \(s:H \to G \rtimes H\) by \(s(h) = (1,h)\), then it is easy to check that \(s\) is a injective hormomorphism such that \(\pi \circ s = id_H\), it shows that the sequence splits, and \(H\) can be identified as \(s(H)\).
\end{proof}

\subsection{Decomposition of endomorphisms}


\begin{colortheorem}[Real-Polar decomposition] $ \\$
For any real endomorphism \(A \in \mathrm{GL}_n( \rr)\), there exists a unique orthogonal endomorphism \(O \in O(n)\) and a unique symmetric positive definite endomorphism \(P \in S^{++}_n(\rr)\) such that
\[A = OP\]
\end{colortheorem}

\begin{proof}
    Here is the outline of the proof:\\
    (1) A \textbf{lemma} about the existence of square root are needed here: If \(A \in S^+_n(\rr)\), then there exists a (unique) \(B \in S^+_n(\rr)\) such that \(B^2 = A\). \\
    (2) Prove that \(A^tA\) is (semi-definite) positive, thus by the lemma there exist a (unique) squar root \(P = \sqrt{A^tA}\), and we prove that \(P\) is positive definite. \\
    (3) Define \(O = AP^{-1}\), and prove that \(O\) is orthogonal.\\
    (4) Prove the uniqueness of the decompoistion.\\

    Here is the details:\\
    (1) By real-spectral theorem, there exists \(O \in O(n)\) such that
    \[A = ODO^T, \quad D = \mathrm{dig}(\lambda_1,...,\lambda_n)\]
    where \(\lambda_i \geq 0\) are the eigenvalues of \(A\), and we set \(\Sigma = \mathrm{dig}(\sqrt{\lambda_1},...,\sqrt{\lambda_n})\) such that
    \[A = (O\Sigma O^T)^2\]
    which shows that \(B = O \Sigma O^T\) is a (unique) square root of \(A\) and it is positive since \(\sqrt{\lambda_i} \geq 0\). 

    (2) \(A^T A\) is clearly symmetric, and it is positive since
    \[x^T (A^T A) x = \|Ax\| \geq 0\]
    for any \(x \in \rr^n\) under the common inner product, and \(A\) is invertible such that \(\|Ax\| = 0\) if and only if \(x = 0\), thus \(A^TA\) is positive definite, so does its square root (the eignevalues is strictly positive).

    (3) \(P^{-1} = (P^{-1})^T\) since \(P\) is symmetric, so we have
    \[O^T O = (P^{-1})^T A^T A P^{-1} = (P^{-1})^T P^2 P^{-1} = I\]

    (4) Suppose that there eixsts another decomposition \(A = O_0P_0\), then we have
    \[A^TA = P_0^T O_0^T O_0 P_0 = P_0^2\]
    so \(P_0\) is the positive square root of \(A^TA\), if the uniqueness of (1) is proved, we can finish the proof, but here we give another proof of the uniqueness only under the existence of the square root: we have \(O_0^{-1}O = P_0P^{-1}\), so it is easy to check that \(P_0P^{-1}\) is \textbf{orthogonal}; notice that \(P_0\) and \(A^TA\) commute, and there exists a polynomial \(f\) such that
    \[f(A^TA) = P\]
    it can be done by using lagrange interpolation, thus \(P_0\) and \(P\) commute as two symmetric matrix, then \(P\) and \(P_0\) can be diagonalized simultaneously by one orthogonal \(H\) such that
    \[P = HDH^T \quad P_0 = H D_0 H^T\]
    where \(D = \mathrm{dig}(\lambda_1,...,\lambda_n)\) and \(D_0 = \mathrm{dig}(\mu_1,...,\mu_n)\), thus
    \[P_0P^{-1} = HD_0^{-1}DH^T\]
    it is \textbf{symmetric and positive definite} since \(D^{-1}_0 D\) is digonal with positive entries, thus \(P_0P^{-1} \in O(n) \cap S^{++}_n(\rr) = \{I\}\), which implies that \(P_0 = P\) and \(O_0 = O\).
\end{proof}

\subsection{Lattice}

\begin{definition}
    Let \(V\) be a \(\rr\)-vector space of dimension \(n\)
\end{definition}

\begin{definition}
    Let \(V\) be a \(\rr\)-vector space of dimension \(n\), a parallelotope of \(V\) with respect to a basis \(\{v_1, \dots, v_n\}\) is the set of points
    \[P = \{\sum_{i = 1}^n c_iv_i \mid c_i \in [0,1)\}\]
    The volume of \(P\) corresponds to the an inner product \(\langle\cdot,\cdot\rangle\) is defined by
    \[\mathrm{Vol}(P) = \sqrt{\det G(v_1,\dots,v_n)}\]
    where \(G(v_1,\dots,v_n)\) is the Gram matrix with entries \(G_{ij} = \langle v_i, v_j \rangle\).
\end{definition}

The definition is resonable with the following reasons:
\begin{enumerate}
    \item Let \(P\) be of dimension 3 with respect to the euclidean inner product, then the volume of \(P\) is given by the triple product
    \[\mathrm{Vol}(P) = |v_1\cdot(v_2\times v_3)| = |\det [v_1, v_2, v_3]|\]
    with respect to the standard basis.

    \item Let \(\{v_i\}_i\) be an orthogonal basis, then the Gram matrix \(G = I_n\) exactly, hence the volume of a hypercube is 1, which same as what we do in plane geometry and solid geometry.
    
    \item 
\end{enumerate}

with the convention above, it allows us to construct the invrariant measure on any real inner product space.

\begin{proposition}
    For any real inner product space \(V\) of dimension \(n\), there exists a natural isometry (\(\rr^n\) equipped with the standard inner product) 
    \[\Phi: \rr^n \to V, \quad (x_1,\dots,x_n) \mapsto \sum_{i=1}^n x_ie_i\]
    where \(\{e_1,\dots,e_n\}\) is an orthonormal basis of \(V\), then \(\mu = \lambda \circ \Phi^{-1}\) defines a Haar measure on \(V\) such that for any parallelotope \(P\) we have
    \(\mu(P) = \mathrm{Vol}(P) \) exactly.
\end{proposition}

\begin{proof}
    \(\mu\) is exactly a pushforward measure, to verify that it is a Haar measure, we need to check the translation invariance: for any \(v \in V\) and any measurable set \(S \subset V\), we have
    \[\mu(v+S) = \lambda(\Phi^{-1}(v+S)) = \lambda(\Phi^{-1}(v) + \Phi^{-1}(S)) = \lambda(\Phi^{-1}(S)) = \mu(S)\]
    where the second equality is due to the linearity of \(\Phi\) and the third equality is due to the translation invariance of the Lebesgue measure \(\lambda\). Now we consider the parallelotope \(P\) with respect to a basis \(\{v_1,\dots,v_n\}\) and set \(\Phi(a_i) = v_i\), then we have
    \[\mu(P) = \lambda (\Phi^{-1}(P)) = |\det A| = \sqrt{\det A^tA}\]
    where \(A\) is the matrix with columns \(a_i\), and isometry shows that \(A^TA\) is exactly the Gram matrix \(G(v_1,\dots,v_n)\), which finishes the proof.
\end{proof}

\section{Representation of groups}

The representation theory appears in almost all branches of mathematics, but the basic ideas and techniques are quite similar, among them the representation theory of finite groups is the most basic one to start with.

    such a objects is called a \textbf{\(k\)-representation of \(G\)}.

    \item A morphism from \((V,\rho)\) to \((W,\tau)\) is a \(k\)-linear map \(T \in \mathcal{L}(V,W)\) such that for any \(g \in G\) we have the following commutative diagram:

Usually, we will denote the set of morphisms from \((V,\rho)\) to \((W,\tau)\) by \(\mathrm{Hom}_G(V,W)\), and the set of endomorphisms of \((V,\rho)\) by \(\mathrm{End}_G(V)\).

It is the basic definition of representation of groups, so firstly we need to describle more structures of the objects and morphisms such that we can know more about what the category exactly looks like.

\begin{definition}
    Let \(W \subset V\) as a subspace, we say that \(W\) is invariant under the representation \((V,\rho)\) if for any \(g \in G\) we have
    \[\rho(W) \subset W\]
    and we define \((W, \rho|_{W})\) as a subrepresentation of \((V,\rho)\). There always exists two trivial subrepresentations: \((\{0\},\rho|_{\{0\}})\) and \((V,\rho)\) itself.
\end{definition}

\begin{remark}
    The definition is exactly natural, in the sense of morphism, subrepresentation is just a representation can be embedded into a larger representation by a injective intertwiner, i.e. if \((W,\tau)\) a representation with \(W \subset V\) as a subspace, then the inclusion \(i: V \to W\) will be an intertiwner such that
    \[\tau(w) = \tau(g)(i(w)) = i(\rho(w)) = \rho(w)\]
    by the definition of \(\tau\), then we can recover the definition of invariant.
\end{remark}

\begin{definition}
    A representation \((V,\rho)\) is called irreducible if it has no non-trivial subrepresentation, i.e. the only subrepresentations are \((\{0\},\rho|_{\{0\}})\) and \((V,\rho)\) itself.
\end{definition}

\begin{colortheorem}[Schur] $ \\$
    Let \((V,\rho)\) and \((W,\pi)\) be two irreducible representations of \(G\), and \(T:V \to W\) be an interwiner, then \(T\) is either an isomorphism or the zero map.
\end{colortheorem}

\begin{proof}
    By above lemma, \(\ker T\) is an invarant subspace of \(V\), so we can conclude that \(\ker T\) is either \(\{0\}\) or \(V\) since \((V,\rho)\) is irreducible. If \(\ker T = V\), then \(T\) is the zero map; If \(\ker T = \{0\}\), then \(T\) is injective, notice that \(\mathrm{im}T\) is an invariant subspace of \(W\) and \((W,\pi)\) is irreducible, in this case we have \(\mathrm{im}T = W\) since \(T\) is injective, thus \(T\) is an isomorphism.
\end{proof}

\begin{remark}
    If two representations are not isomorphic, then intwerwiner must be zero, i.e. \(\mathrm{Hom}_{G}(V,W) = \{0\}\), and we can define a equivalence relation on \textbf{the irreducible repesentations} of \(G\) as following:
    \[(V,\rho) \sim (W,\pi) \iff \text{they are isomorphic}\]
    and we use \(\hat{G}\) to denote the set of equivalence classses.
\end{remark}

\begin{colorcorollary}

\end{colorcorollary}

\begin{proof}
    \(\End_G(V)\) contains zero map and isomorphisms by schur's lemma, thus it is a divison algebra over \(k\), and the unique finite-dimensional division algebra over \(\cc\) is \(\cc\) itself, so we have \(\End_G(V) \cong \cc\).
\end{proof}

\subsection{Group algebra and convolution}

\begin{definition}
    Let \(G\) be a group and \(k\) be a field, then the group alegbra \(k[G]\) is a \(k\)-algebra defined as following:
    \[k[G] : = \{\sum_{g \in G}c_ge_g \mid c_g \in k \text{ and } c_g = 0 \text{ for almost all } g\}\]
    where \(e_g\) is a formal component with respect to \(g\), which satisifies
    \[\forall g,h \in G, \quad e_g \times e_h = e_{gh}\]
    and \(e_1\) is the multiplicative identity of \(k[G]\),
    the operation is defined as following:\\
    - Addition:
    \[\sum_{g \in G}a_ge_g + \sum_{g \in G}b_ge_g = \sum_{g \in G} (a_g + b_g)e_g\]
    - Scalar multiplication:
    \[c \cdot \sum_{g \in G} a_ge_g = \sum_{g \in G} (ca_g)e_g\]
    - Multiplication:
    \[(\sum_{g \in G} a_ge_g) \times (\sum_{h \in G} b_he_h) = \sum_{g \in G}c_ge_g\]
    with \(c_g = \sum_{uv = g}a_ub_v\).
\end{definition}

\begin{remark}
    Generally, the definition can be extended to a monoid and a ring insetead of a group and a field. If \(G\) is a finite group, then \(k[G]\) is a finite-dimensional \(k\)-vector space with basis \(\{e_g\}_{g \in G}\). Moreover, the group alegbra is a commutative algebra if and only if \(G\) is an abelian group.
\end{remark}

\textcolor{blue}{More generally, any group action can be extended linearly to be a reprsentation}, we construct as following
\begin{enumerate}
    \item Let \(G \circlearrowleft X\) be a group action with a well-defined operation \(g*x\), firstly we can extend the set \(X\) to be a vector space over \(k\)

    where \(e_x\) is a formal component with respect to \(x\), and toghether forms a basis of \(k[G]\), it is easy to check that it is a vector space.

    \[\rho(g)(\sum_{x \in X}c_xe_x) = \sum_{x \in X}c_x e_{g*x}\]

& X \arrow[d, "\iota"] \\
k[X] \arrow[r, "\rho(g)"] 
& k[X]
\end{tikzcd}
\]
    which shows that \(\rho\) is exactly the linear extension of \(\sigma\) as a representation of \(G\).

    \item The extension can be applied to morphism of \(G\)-sets. if \(G\) is a group acting on two sets \(X\) and \(Y\) with opreation \(*\) and \(\star \) respectively, then we consider a \(G\)-map \(f: X \to Y\) which satisfies
    \[\forall g \in G, x \in X,\quad f(g*x) = g \star f(x)\]
    then we can extend \(f\) to be a linear map \(\tilde{f}: k[X] \to k[Y]\) by
    \[\tilde{f}(\sum_{x \in X}c_xe_x) = \sum_{x \in X}c_x e_{f(x)}\]

    &= \sum_{x \in X}c_x e_{g \star f(x)})\\
    &= \sum_{x \in X}c_x e_{f(g*x)})\\
    &= \tilde{f}(\sum_{x \in X}c_x e_{g*x})\\
    &= \tilde{f} \circ \rho_X(g)(\sum_{x \in X}c_xe_x)  
    \end{align*}
    which shows that \(\tilde{f}\) is an interwiner between \(\rho_X\) and \(\rho_Y\).
\end{enumerate}

The constuction shows that representation thoery is a powerful tool, as a core tool in group theory, group action can be contained in represetation theory, which can be concluded by the language of category as following:

    - On morphisms: \(L(f) = \tilde{f}\) where \(\tilde{f}\) is the linear extension of the \(G\)-map \(f\).\\

Now we turn our attention to the finite-dimensional representation of finite groups  


\begin{definition}
    Let \(G\) be a finite group and \(k\) be a field, we set the space of functions on \(G\) by \(\mathcal{F}(G)\), and we define the operation such that it is a \(k\)-alegbra:\\
    - Addition:
    \[(f+g)(x) = f(x) + g(x)\]
    - Scalar multiplication:
    \[(c\cdot f)(x) = cf(x)\]
    - Multiplication:
\[(fg)(x) = f(x) \cdot g(x)\]
    where \(f,g \in \mathcal{F}(G)\) and \(c \in k\).
\end{definition}

As a vector space, \(\mathcal{F}(G)\) is finite-dimensional with basis \(\{\delta_g\}_{g \in G}\) where \(\delta_g\) is the delta function defined by
\[\delta_g(x) = \begin{cases} 1 & x = g\\ 0 & x \neq g\end{cases}\]
hence there exists a natural isomorphism \(\mathcal{F}(G) \cong k[G]\) by sending \(\delta_g\) to \(e_g\).

However, as \(k\)-alegbra the group alegbra \(k[G]\) is not isomorphic to \(\mathcal{F}(G)\), since \(k[G]\) can be non-commutative if \(G\) while \(\mathcal{F}(G)\) is always commutative, but a different multiplication can be defined on \(\mathcal{F}(G)\) such that we can get an isomorphism.

\subsection{Character and orthogonality relations}

\begin{definition}
    Suppose that \((V,\rho)\) is a finite-dimensional representation of \(G\), we define the character of \((V,\rho)\) as a function \(\chi_\rho: G \to k\) by
    \[\chi_\rho(g) = \mathrm{tr}(\rho(g))\]
    for any \(g \in G\).
\end{definition}

    Notice that if \(\dim V = 1\), then \(\mathrm{GL}(V) \cong k^\times\), which shows that the character of the representation will be the representation itself: a group homomorphism from \(G\) to \(k^\times\). \textcolor{blue}{Generally, the character of a representation is not necessary a group homomorphism.}


\begin{definition}
    Let \(G\) be a group and \(k\) be a field, we denote \(\cl(G) = G/\sim_c\) be the quotient of \(G\) by the conjugacy relation
    \[x \sim y \iff \exists g \in G, y = gxg^{-1}\]
    a function \(f:\cl(G) \to k\) is called a \textbf{class function}, or equivalently, a function \(f:G \to k\) such that 
    \[f(gxg^{-1}) = f(x)\]
    for any \(g,x \in G\).
\end{definition}

\begin{remark}
    The algebraic properties of trace shows that \textcolor{blue}{the character of a representation is always a class function}, thus it can be regarded as a function on \(\cl(G)\).

    The conjugacy classes can be described by group action \(G \circlearrowleft G\) with operation \(g*x = gxg^{-1}\), then \(\cl(G)\) is exactly the set of orbits with 
    \[C_G(x) = \{g \in G \mid gxg^{-1}= x\}\]
    the centralizer (stablizer) of \(x\) in \(G\). When \(G\) is finite, \([G:C_G(x)]\) is exactly the size of orbit of \(x\).
\end{remark}

\begin{proposition}
        Let \((V,\rho)\) be a finite-dimensional representation of \(G\), then \\
        (a) \(\chi_\rho(e) = \dim V\).\\ 
        (b) \(\chi_{\rho^*}(g) = \chi_\rho(g^{-1}) = \mathrm{tr}(\rho(g)^{-1})\) for any \(g \in G\).\\
        Let \((W,\pi)\) be another finite-dimensional representation of \(G\), then \\
        (c) \(\chi_{\rho \oplus \pi} = \chi_\rho + \chi_\pi\)\\
        (d) \(\chi_{\rho \otimes \pi} = \chi_{\rho} \cdot \chi_\pi\)
\end{proposition}
